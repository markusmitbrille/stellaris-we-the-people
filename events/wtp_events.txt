# on monthly pulse
country_event = {
	id = wtp.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		# only regular empires
		is_country_type = default
		NOT = {
			has_ethic = ethic_gestalt_consciousness
		}
	}
	immediate = {
		every_owned_planet = {
			if = {
				limit = {
					sector = {
						is_same_value = PREVPREV.capital_scope.sector
					}
				}
				wtp_effect_remove_unrest = yes
				wtp_effect_remove_vanilla_unrest = yes
				set_variable = {
					which = wtp_var_unrest
					value = 0
				}
			}
			else = {
				planet_event = {
					id = wtp.100
					days = 1
					random = 29
				}
			}
		}
	}
}

# monthly unrest update
planet_event = {
	id = wtp.100
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		wtp_can_have_unrest = yes
	}
	immediate = {
		# calculate unclamped trend
		wtp_effect_calculate_planet_trend = yes
		# change unrest by clamped trend
		if = {
			limit = {
				check_variable = {
					which = wtp_var_trend
					value < -1
				}
			}
			change_variable = {
				which = wtp_var_unrest
				value = -1
			}
		}
		else_if = {
			limit = {
				check_variable = {
					which = wtp_var_trend
					value > 1
				}
			}
			change_variable = {
				which = wtp_var_unrest
				value = 1
			}
		}
		else = {
			change_variable = {
				which = wtp_var_unrest
				value = wtp_var_trend
			}
		}
		# clamp unrest
		if = {
			limit = {
				check_variable = {
					which = wtp_var_unrest
					value < 0
				}
			}
			set_variable = {
				which = wtp_var_unrest
				value = 0
			}
		}
		# start unrest
		if = {
			limit = {
				check_variable = {
					which = wtp_var_unrest
					value >= @unrest_threshold_high
				}
				NOT = {
					wtp_trigger_has_unrest = yes
				}
			}
			planet_event = {
				id = wtp.200
			}
			break = yes
		}
		# cancel unrest
		if = {
			limit = {
				check_variable = {
					which = wtp_var_unrest
					value <= @unrest_threshold_low
				}
				wtp_trigger_has_unrest = yes
			}
			planet_event = {
				id = wtp.300
			}
			break = yes
		}
		# trigger rebellion
		if = {
			limit = {
				check_variable = {
					which = wtp_var_unrest
					value >= @rebellion_threshold
				}
				wtp_trigger_has_unrest = yes
			}
			planet_event = {
				id = wtp.400
			}
			break = yes
		}
	}
}

		}
	}

	}
